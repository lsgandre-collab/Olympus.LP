<!-- SCRIPTS (3D, GSAP, toggles) -->
<script>
  // Theme toggle
  function toggleTheme() {
    document.documentElement.dataset.theme = document.documentElement.dataset.theme === 'dark' ? 'light' : 'dark';
  }

  // Language toggle (simples, expanda se precisar mais textos)
  let lang = 'pt';
  function toggleLanguage() {
    lang = lang === 'pt' ? 'en' : 'pt';
    document.getElementById('lang-btn').textContent = lang.toUpperCase();
    document.querySelectorAll('[data-en]').forEach(el => {
      const en = el.getAttribute('data-en');
      const pt = el.dataset.pt || el.textContent.trim();
      el.textContent = lang === 'en' ? en : pt;
      if (!el.dataset.pt) el.dataset.pt = pt;
    });
  }

  // GSAP fade-ups
  gsap.registerPlugin(ScrollTrigger);
  document.querySelectorAll('[data-gsap="fade-up"]').forEach((el, i) => {
    gsap.from(el, {
      y: 100, opacity: 0, duration: 1.2, ease: "power3.out",
      scrollTrigger: { trigger: el, start: "top 85%" },
      delay: i * 0.15
    });
  });

  // Three.js 3D hero (CEO central + agentes orbitando)
  const scene = new THREE.Scene();
  const camera = new THREE.PerspectiveCamera(50, window.innerWidth / window.innerHeight, 0.1, 1000);
  const renderer = new THREE.WebGLRenderer({ canvas: document.getElementById('three-canvas'), alpha: true });
  renderer.setSize(window.innerWidth, window.innerHeight);
  camera.position.z = 45;

  const light = new THREE.PointLight(0xffffff, 2.5, 100);
  light.position.set(0, 20, 30);
  scene.add(light);

  // CEO (esfera dourada central)
  const ceo = new THREE.Mesh(new THREE.SphereGeometry(5, 64, 64), new THREE.MeshPhongMaterial({color: 0xd4af77, emissive: 0xb45309, shininess: 100}));
  scene.add(ceo);

  // Agentes orbitando (orbes coloridos)
  const colors = [0xef4444, 0x22c55e, 0x3b82f6, 0xd97706, 0xec4899, 0x8b5cf6];
  for (let i = 0; i < 6; i++) {
    const orb = new THREE.Mesh(new THREE.SphereGeometry(2, 32, 32), new THREE.MeshPhongMaterial({color: colors[i], emissive: colors[i], emissiveIntensity: 0.8}));
    orb.userData = { angle: i * Math.PI * 2 / 6, radius: 22 };
    scene.add(orb);
  }

  function animate() {
    requestAnimationFrame(animate);
    const t = Date.now() * 0.0004;
    scene.children.forEach(c => {
      if (c.userData.angle !== undefined) {
        const a = c.userData.angle + t * 1.2;
        c.position.x = Math.cos(a) * c.userData.radius;
        c.position.z = Math.sin(a) * c.userData.radius * 0.8;
        c.position.y = Math.sin(a * 2) * 8;
      }
    });
    ceo.rotation.y += 0.004;
    renderer.render(scene, camera);
  }
  animate();

  window.addEventListener('resize', () => {
    camera.aspect = window.innerWidth / window.innerHeight;
    camera.updateProjectionMatrix();
    renderer.setSize(window.innerWidth, window.innerHeight);
  });
</script>
